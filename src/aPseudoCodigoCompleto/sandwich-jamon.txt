HacerElPan ()
{
	Plantar el trigo
	Cuidar la siembra
	Cosechar el trigo
	Hacer harina
	Añadir agua y levadura
	Hornear el Pan
}

HacerElJamon ()
{
	Criar los cerdos
	Sacrificar los Cerdos
	Hacer jamon
}

HacerLaCrema ()
{
	Ordeñar la vaca
	Hacer Crema
}

ConseguirTomate ()
{
	Plantar el tomate
	Cuidar la siembra
	Cosechar el tomate
	Lavar el tomate
	Desinfectar el tomate
	Rebanar un tomate
}

ConseguirLaLechuga ()
{
	Plantar la lechuga
	Cuidar la siembra
	Cosechar la lechuga
	Lavar la lechuga
	Desinfectar la lechuga
	Partir la lechuga
}

TomarElPan ()
{
	HacerElPan ()
	Tomar el cuchillo
	Cortar el pan en rebanadas
	Dejar el cuchillo
	Tomar dos rebanadas
}

RebanadasDeJamon ()

}
